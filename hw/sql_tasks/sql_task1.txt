8)Повысить возраст на 2 года всем актрисам (female) старше 40 лет.

9)Вывести актрис из Франции младше 35 лет.
SELECT name, last_name FROM actors

10)Вывести 5 самых молодых актёров.
SELECT * FROM actors
WHERE age IS NOT NULL AND sex = "male"
ORDER BY age ASC 
LIMIT 5

11)Посчитать и вывести общее количество актёров, средний возраст, минимальный и максимальный возраст
SELECT COUNT(name) AS "Actors count",
AVG(age) as "Average age",
MIN(age) as "Minimal age",
MAX(age) as "Maximal age"
FROM actors

12)Вывести страны, в которых более 5 и менее 10 актёров.
SELECT country, COUNT(*) AS "actor count"
FROM actors
GROUP BY country
HAVING "actor count" > 5 AND "actor count" < 10

13)Посчитать средний возраст мужчин и женщин.
SELECT sex, avg(age) AS "Average age"
FROM actors
WHERE sex is not null and age is not null
GROUP BY sex

14)Для каждой страны вывести самого молодого актёра.
SELECT country, name, last_name, min(age) AS "Average age"
FROM actors
WHERE country != "Unknown" AND sex = "male"
GROUP BY country

15)Вывести страны, в которых нет ни одной актрисы (female).
SELECT country
FROM actors
WHERE country != "Unknown" AND sex IS NOT NULL
GROUP BY country
HAVING COUNT(CASE WHEN sex = "female" THEN 1 END) = 0;

16)Вывести актёров, которые являются самыми старшими в своей стране.
SELECT country, name, last_name, Max(age) AS "old"
FROM actors
WHERE country != "Unknown" AND sex = "male"
GROUP BY country

17)Вывести страну с самым большим средним возрастом актёров.

18)Вывести актёров из тех стран, где есть хотя бы одна актриса старше 50 лет.
SELECT country, AVG(age) AS average_age
FROM actors
WHERE age IS NOT NULL AND country IS NOT NULL
GROUP BY country
ORDER BY average_age DESC
LIMIT 1;

20)Найти возраст(а), который встречается более 2 раз.
     Затем вывести всех актёров этого возраста.
SELECT id, name, last_name, age, sex, country
FROM actors
WHERE age IN (
    SELECT age
    FROM actors
    WHERE age IS NOT NULL
    GROUP BY age
    HAVING COUNT(*) >= 2
)
ORDER BY age, name;
